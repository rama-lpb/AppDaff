- puis on modifie le seeder pour quavant la connexion a la base on se connecte a cloudinary  en lui cr√©ant une nouvelle instance :
            <?php

            use Cloudinary\Cloudinary;
            use Cloudinary\Configuration\Configuration;
            use Dotenv\Dotenv;


            require_once __DIR__ . '/../vendor/autoload.php';
            $cloud = require __DIR__ . '/../app/config/cloudinary.php';


            Configuration::instance([
                'cloud' => [
                    'cloud_name' => $cloud['cloud_name'],
                    'api_key'    => $cloud['api_key'],
                    'api_secret' => $cloud['api_secret'],
                ],
                'url' => ['secure' => true]
            ]);

            $cloudinary = new Cloudinary(Configuration::instance());
